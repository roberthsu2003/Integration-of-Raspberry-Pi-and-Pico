    讀取 Pico 內建溫度感測器的溫度值
    
    Returns:
        float: 溫度值（攝氏度）
    """
    sensor_temp = machine.ADC(4)
    reading = sensor_temp.read_u16() * (3.3 / 65535)
    temperature = 27 - (reading - 0.706) / 0.001721
    return temperature
